query Genes(
  $entrez: [Int!],
  $geneType: [String!]
  $sample: [String!]
) {
  genes(
    entrez: $entrez
    geneType: $geneType
    sample: $sample
  ){
    entrez
    hgnc
    description
    friendlyName
    ioLandscapeName
    geneFamily
    geneFunction
    immuneCheckpoint
    pathway
    superCategory
  }
}
